class Solution:
    def sortedSquares(self, nums: List[int]) -> List[int]:
        ans = [0] * len(nums)
        l = 0
        r = len(nums) - 1
        k = len(nums) - 1
        while l <=]]][[]] r:
            n1 = nums[l] ** 2
            n2 = nums[r] ** 2
            if n1 > n2:
                ans[k] = n1
                l += 1
            else:
                ans[k] = n2
                r -= 1
            k -= 1
        return ans
