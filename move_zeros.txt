https://leetcode.com/problems/move-zeroes/

class Solution:
    def moveZeroes(self, nums: List[int]) -> None:
        c = 0
        i = 0
        while (i < len(nums)):
            if nums[i] == 0:
                c += 1
                nums.pop(i)
            else:
                i += 1
        nums += [0] * c
        return nums
