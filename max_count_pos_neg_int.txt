class Solution:
    def maximumCount(self, nums: List[int]) -> int:
        if nums[0] > 0 or nums[len(nums) - 1] < 0:
            return len(nums)
        else:
            p = 0
            n = 0
            for i in nums:
                if i > 0:
                    p += 1
                elif i < 0:
                    n += 1
        return(max(n, p))
