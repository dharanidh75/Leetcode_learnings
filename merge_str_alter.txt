https://leetcode.com/problems/merge-strings-alternately/

class Solution:
    def mergeAlternately(self, word1: str, word2: str) -> str:
        s = ''
        l = 0
        r = 0
        while l < len(word1) and r < len(word2):
            s += word1[l]
            l += 1
            s += word2[r]
            r += 1
        if l < len(word1):
            s += word1[l:]
        elif r < len(word2):
            s += word2[r:]
        return s
