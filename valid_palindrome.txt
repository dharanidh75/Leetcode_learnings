class Solution:
    def isPalindrome(self, s: str) -> bool:
        r = ''
        s = s.lower()
        s = s.replace(" ", "")
        for i in range(len(s)):
            if (s[i] >= 'a'and s[i] <= 'z') or (s[i] >= "0" and s[i] <= "9"):
                r += s[i]
            else:
                continue
        if r == r[::-1]:
            return True
        else:
            return False
