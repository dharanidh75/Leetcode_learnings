https://leetcode.com/problems/find-the-difference/

class Solution:
    def findTheDifference(self, s: str, t: str) -> str:
        l = 0
        r = 0
        while l < len(s) and r < len(t):
            if t[r] not in s:
                return t[r]
            l += 1
            r += 1
        return t[r]
