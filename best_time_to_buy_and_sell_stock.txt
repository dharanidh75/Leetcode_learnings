class Solution:
    def maxProfit(self, prices: List[int]) -> int:
        min_profit = float('inf')
        max_profit = 0
        for i in range(len(prices)):
            if prices[i] < min_profit:
                min_profit = prices[i]
            else:
                if prices[i] - min_profit > max_profit:
                    max_profit = prices[i] - min_profit
        return max_profit
